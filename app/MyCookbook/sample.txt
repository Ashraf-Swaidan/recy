import React, { Suspense } from 'react';
import RecipeCard from '../components/recipe/RecipeCard';
import { currentUser } from '@clerk/nextjs/server';
import { Book, Search, Filter } from 'lucide-react';
import { RecipeWithOwner } from '@/app/lib/types';

// Loading component for Suspense
const LoadingGrid = () => (
  <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
    {[...Array(8)].map((_, i) => (
      <div 
        key={i} 
        className="h-64 rounded-lg bg-gray-200 animate-pulse"
      />
    ))}
  </div>
);

async function getCookbookRecipes(userId: string): Promise<RecipeWithOwner[]> {
  // Fetch cookbook data
  const cookbookRes = await fetch(
    `${process.env.NEXT_PUBLIC_BASE_URL}/api/users/${userId}/cookbook`,
    { next: { revalidate: 60 } } // Cache for 1 minute
  );

  if (!cookbookRes.ok) {
    if (cookbookRes.status === 404) return [];
    throw new Error('Failed to fetch cookbook');
  }

  const cookbookData = await cookbookRes.json();
  if (!cookbookData.savedRecipes?.length) return [];

  // Fetch all recipes in parallel
  const recipes = await Promise.all(
    cookbookData.savedRecipes.map(async (savedRecipe: any) => {
      const recipeRes = await fetch(
        `${process.env.NEXT_PUBLIC_BASE_URL}/api/recipe/${savedRecipe.recipeId.toString()}`,
        { next: { revalidate: 60 } }
      );
      if (!recipeRes.ok) {
        throw new Error(`Failed to fetch recipe ${savedRecipe.recipeId}`);
      }
      return recipeRes.json();
    })
  );

  return recipes;
}

const RecipeGrid = async ({ userId }: { userId: string }) => {
  const recipes = await getCookbookRecipes(userId);

  if (!recipes.length) {
    return (
      <div className="min-h-screen flex flex-col items-center justify-center p-8 bg-gray-50">
        <Book className="w-16 h-16 text-orange-600 mb-4" />
        <h2 className="text-2xl font-semibold text-gray-800 mb-2">Your Cookbook is Empty</h2>
        <p className="text-gray-600 text-center max-w-md">
          Start building your personal cookbook by saving recipes you love!
        </p>
      </div>
    );
  }

  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
      {recipes.map((recipe) => (
        <RecipeCard key={recipe._id.toString()} recipe={recipe} />
      ))}
    </div>
  );
};

const CookbookPage = async () => {
  const user = await currentUser();
  const userId = user?.publicMetadata.userId as string;

  return (
    <div className="min-h-screen bg-gray-50 p-8">
      {/* Header Section */}
      <div className="max-w-7xl mx-auto mb-8 border-b-2">
        <div className="flex flex-col sm:flex-row sm:items-center sm:justify-between gap-4 mb-6">
          <div>
            <h1 className="text-3xl font-bold text-gray-900 mb-2">My Cookbook</h1>
          </div>
          
          {/* Search Bar */}
          <div className='flex flex-col sm:flex-row gap-4'>
            <div className="relative w-full sm:w-auto">
              <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400 w-5 h-5" />
              <input
                type="search"
                placeholder="Search your recipes..."
                className="w-full sm:w-64 pl-10 pr-4 py-2 border border-gray-200 rounded-lg focus:outline-none focus:ring-2 focus:ring-orange-500"
              />
            </div>
            <button className="flex items-center gap-2 px-4 py-2 border border-gray-200 rounded-lg hover:bg-gray-50">
              <Filter className="w-5 h-5 text-gray-600" />
              <span>Filter</span>
            </button>
          </div>
        </div>
      </div>

      {/* Recipe Grid with Suspense */}
      <div className="max-w-7xl mx-auto">
        <Suspense fallback={<LoadingGrid />}>
          <RecipeGrid userId={userId} />
        </Suspense>
      </div>
    </div>
  );
};

export default CookbookPage;
